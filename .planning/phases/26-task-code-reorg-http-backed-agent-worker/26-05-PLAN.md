```python
"""Contract tests for GET /api/internal/agent/whoami (Phase 26 D-15..D-17)."""

from __future__ import annotations

from datetime import datetime, timezone

from httpx import AsyncClient
import pytest
from sqlalchemy import update
from sqlalchemy.ext.asyncio import AsyncSession

from phaze.models.agent import Agent


async def test_whoami_happy_path(authenticated_client: AsyncClient, seed_test_agent: tuple[Agent, str]) -> None:
    """Authenticated GET /whoami returns 200 + AgentIdentity body matching the seeded agent."""
    agent, _ = seed_test_agent
    response = await authenticated_client.get("/api/internal/agent/whoami")
    assert response.status_code == 200, response.text
    body = response.json()
    assert body["agent_id"] == agent.id
    assert body["name"] == agent.name
    assert body["scan_roots"] == agent.scan_roots
    # created_at is ISO-8601 with timezone
    parsed = datetime.fromisoformat(body["created_at"])
    assert parsed.tzinfo is not None


async def test_whoami_missing_header_returns_401(client: AsyncClient) -> None:
    """No Authorization header -> 401 (HTTPBearer auto_error)."""
    response = await client.get("/api/internal/agent/whoami")
    assert response.status_code == 401
    # WWW-Authenticate: Bearer per RFC 6750 (Phase 25 D-06)
    assert "bearer" in response.headers.get("WWW-Authenticate", "").lower()


async def test_whoami_unknown_token_returns_403(client: AsyncClient) -> None:
    """Well-formed bearer with unknown hash -> 403 (Phase 25 D-06)."""
    response = await client.get(
        "/api/internal/agent/whoami",
        headers={"Authorization": "Bearer phaze_agent_unknown-token-string-1234"},
    )
    assert response.status_code == 403


async def test_whoami_revoked_token_returns_403(
    client: AsyncClient,
    session: AsyncSession,
    seed_test_agent: tuple[Agent, str],
) -> None:
    """Revoking an agent immediately blocks future /whoami calls (Phase 25 AUTH-04)."""
    agent, raw_token = seed_test_agent
    # Confirm the token is valid first
    pre = await client.get(
        "/api/internal/agent/whoami",
        headers={"Authorization": f"Bearer {raw_token}"},
    )
    assert pre.status_code == 200
    # Revoke
    await session.execute(
        update(Agent).where(Agent.id == agent.id).values(revoked_at=datetime.now(timezone.utc))
    )
    await session.commit()
    # Next call must fail with 403, no restart required
    post = await client.get(
        "/api/internal/agent/whoami",
        headers={"Authorization": f"Bearer {raw_token}"},
    )
    assert post.status_code == 403
```

```python
"""GET /api/internal/agent/whoami -- agent identity probe (Phase 26 D-15..D-17).

Returns the calling agent's identity (id, name, scan_roots, created_at) so the
agent worker can verify (a) its bearer token is valid and (b) the token-derived
agent_id matches the operator-supplied PHAZE_AGENT_QUEUE env var at startup
(Plan 10 anti-misconfiguration probe, RESEARCH Pitfall 1).

Phase 29 will also use this endpoint for the Agents admin page's reachability probe.
"""

from typing import Annotated

from fastapi import APIRouter, Depends, status

from phaze.models.agent import Agent
from phaze.routers.agent_auth import get_authenticated_agent
from phaze.schemas.agent_identity import AgentIdentity


router = APIRouter(prefix="/api/internal/agent/whoami", tags=["agent-internal"])


@router.get("", status_code=status.HTTP_200_OK, response_model=AgentIdentity)
async def whoami(
    agent: Annotated[Agent, Depends(get_authenticated_agent)],
) -> AgentIdentity:
    """Return the calling agent's identity.

    Per D-15: response body is `{agent_id, name, scan_roots, created_at}`.
    No request body. agent_id is NEVER derived from request body (AUTH-01); it
    comes from the auth dep's token lookup.

    401: missing/malformed Authorization header (handled by HTTPBearer auto_error
         in get_authenticated_agent's dep chain).
    403: well-formed bearer whose hash is unknown OR whose row has
         revoked_at IS NOT NULL (handled inside get_authenticated_agent).
    200: success.
    """
    return AgentIdentity(
        agent_id=agent.id,
        name=agent.name,
        scan_roots=agent.scan_roots,
        created_at=agent.created_at,
    )
```
